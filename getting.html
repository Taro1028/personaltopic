window.addEventListener('load', () => { const amountTWD = document.getElementById("amountTWD"); const amountUSD = document.getElementById("amountUSD"); // 固定匯率 1 TWD = 0.031 USD const exchangeRate = 0.031; // 台幣換美金 function convertCurrency() { const amount = amountTWD.value; if (amount === "" || isNaN(amount) || amount <= 0) { amountUSD.value = ""; return; } // 計算結果四捨五入到小數點後兩位 amountUSD.value = (amount * exchangeRate).toFixed(2); // 監聽輸入變更 let now = new Date() let hour = ('0' + now.getHours()).slice(-2) let minute = ('0' + now.getMinutes()).slice(-2) document.getElementsByClassName('timestamp')[0].innerHTML = ` ${now.toLocaleDateString()} ${hour}:${minute} ` } amountTWD.addEventListener("input", convertCurrency); })
